CREATE TABLE Customer (
    customer_id INTEGER PRIMARY KEY,
    customer_name VARCHAR(255),
    customer_email VARCHAR(255),
    customer_phone VARCHAR(15),
    customer_address VARCHAR(255)
);

CREATE TABLE Employee (
    employee_id INTEGER PRIMARY KEY,
    employee_name VARCHAR(255),
    employee_phone VARCHAR(15),
    employee_email VARCHAR(255),
    admin_id INTEGER,
    FOREIGN KEY (admin_id) REFERENCES Admin(admin_id)
);

CREATE TABLE Admin (
    admin_id INTEGER PRIMARY KEY,
    admin_name VARCHAR(255),
    admin_email VARCHAR(255),
    admin_phone VARCHAR(15)
);

CREATE TABLE Order (
    order_id INTEGER PRIMARY KEY,
    customer_id INTEGER,
    employee_id INTEGER,
    order_date DATE,
    order_status VARCHAR(50),
    admin_id INTEGER,
    FOREIGN KEY (customer_id) REFERENCES Customer(customer_id),
    FOREIGN KEY (employee_id) REFERENCES Employee(employee_id),
    FOREIGN KEY (admin_id) REFERENCES Admin(admin_id)
);

CREATE TABLE FoodItem (
    food_item_id INTEGER PRIMARY KEY,
    food_name VARCHAR(255),
    food_category VARCHAR(100),
    food_price DECIMAL(10, 2),
    food_stockquantity INTEGER,
    admin_id INTEGER,
    FOREIGN KEY (admin_id) REFERENCES Admin(admin_id)
);

CREATE TABLE OrderDetail (
    order_detail_id INTEGER PRIMARY KEY,
    order_id INTEGER,
    food_item_id INTEGER,
    orderdetail_quantity INTEGER,
    FOREIGN KEY (order_id) REFERENCES Order(order_id),
    FOREIGN KEY (food_item_id) REFERENCES FoodItem(food_item_id)
);

CREATE TABLE PantryLocation (
    location_id INTEGER PRIMARY KEY,
    location_name VARCHAR(255),
    pantrylocation_section VARCHAR(100)
);

CREATE TABLE FoodItemLocation (
    food_item_id INTEGER,
    location_id INTEGER,
    PRIMARY KEY (food_item_id, location_id),
    FOREIGN KEY (food_item_id) REFERENCES FoodItem(food_item_id),
    FOREIGN KEY (location_id) REFERENCES PantryLocation(location_id)
);
